83 private void emitSingleParticle() {
84 foreach (ref particle; particles) {
85 if (!particle.active) {
86 particle.x = x;
87 particle.y = y;
88 float angle = uniform(-config.spread, config.spread) * (PI / 180.0);
89 float speed = config.speed + uniform(-config.speed * 0.3, config.speed * 0.3);
90 particle.velocityX = cos(angle) * speed;
91 particle.velocityY = -sin(angle) * speed;
92
93 particle.lifetime = 0.0;
94 particle.maxLifetime = config.particleLifetime + uniform(-0.2, 0.2);
95
96 particle.r = cast(ubyte)(config.r + uniform(-20, 20));
97 particle.g = cast(ubyte)(config.g + uniform(-20, 20));
98 particle.b = cast(ubyte)(config.b + uniform(-20, 20));
99 particle.alpha = 255;
100
101 particle.size = uniform(config.minSize, config.maxSize + 1);
102 particle.active = true;
103 break;}

107 void update(double deltaTime) {
108 if (isEmitting) {
109 emissionTimer += deltaTime;
110 float timeBetweenEmissions = 1.0 / config.emissionRate;
111 while (emissionTimer >= timeBetweenEmissions) {
112 emitSingleParticle();
113 emissionTimer -= timeBetweenEmissions;
114 }
115 }
116 foreach (ref particle; particles) {
117 if (!particle.active) continue;
118 particle.lifetime += deltaTime;

120 if (particle.lifetime >= particle.maxLifetime) {
121 particle.active = false;

123 }
124 if (config.useGravity) {

126 }

128 particle.x += particle.velocityX * deltaTime;
129 particle.y += particle.velocityY * deltaTime;

132 particle.alpha = cast(ubyte)(255 * (1.0 - lifeRatio));
133 }
