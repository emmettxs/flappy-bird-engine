161 if (currentLevel is null) return;
162 // skip collision if invincible

173 if (collider !is null && transform !is null) {
174 // check if bird collides with this object
175 bool collision = (
176 birdX < transform.x + collider.width &&
177 birdX + effectiveWidth > transform.x &&
178 birdY < transform.y + collider.height &&
179 birdY + effectiveHeight > transform.y

181
182 if (collision) {
183 if (collider.tag == "pipe" || collider.tag == "ground" || collider.tag == "wall" || collider.tag == "ceiling") {
184 triggerGameOver();

249 private void updatePipes(double deltaTime) {
250 pipeTimer += deltaTime;
251 if (pipeTimer >= pipeSpawnRate) {
252 spawnPipe();
253 pipeTimer = 0.0;
254 }
255 for (size_t i = 0; i < pipes.length; i++) {
256 pipes[i].x += cast(int)(pipeSpeed * deltaTime);
257 if (!pipes[i].scoredPoint && pipes[i].x + pipeWidth < birdX) {
258 pipes[i].scoredPoint = true;
259 score++;
260 writeln("Score: ", score);
261 }
262 }
263 Pipe[] newPipes;
264 foreach (pipe; pipes) {
265 if (pipe.x + pipeWidth > 0) {
266 newPipes ~= pipe;
267 }
268 }
269 pipes = newPipes;
